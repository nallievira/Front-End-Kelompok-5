```javascript
//Sistem musik latar dengan transisi seamless menggunakan dual audio player
let audioPlayer = null;
let nextAudioPlayer = null;
let currentSongIndex = 0;
let isTransitioning = false;

const musicTracks = [
  "assets/music/O Ina Ni Keke.mp3",
  "assets/music/videoplayback.mp3",
];

//Fungsi untuk memulai transisi musik seamless
function startSeamlessTransition() {
  if (isTransitioning) return;

  isTransitioning = true;
  const currentPlayer = audioPlayer;
  const nextPlayer = nextAudioPlayer;

  //Mulai lagu berikutnya sebelum lagu saat ini selesai
  nextPlayer.play().then(() => {
    setTimeout(() => {
      currentPlayer.pause();
      //Tukar/ganti lagu saat ini dengan lagu berikutnya
      audioPlayer = nextPlayer;
      nextAudioPlayer = currentPlayer;
      currentSongIndex = (currentSongIndex + 1) % musicTracks.length;
      preloadNextSong();
      isTransitioning = false;
    }, 100);
  });
}
```

function setupScrollAnimation() {
  let hasAppeared = {
    rasa: false,
    manado: false,
    subtitle: false,
    navbar: false,
  };

  $(window).on("scroll", function () {
    const scrollY = $(window).scrollTop();

    //Rasa animation
    if (!hasAppeared.rasa && scrollY >= 10) {
      const progress = Math.min(scrollY / 100, 1);
      $(".screen-title-rasa").css({
        opacity: progress,
        transform: `translateX(-60%) translateY(${30 - progress * 30}px)`,
      });
    }

    //Visibilitas kontrol musik saat user scroll
    const musicControls = document.getElementById("music-controls");
    const introHeight = window.innerHeight;
    if (scrollY > introHeight * 0.8) {
      musicControls.classList.add("hidden");
    } else {
      musicControls.classList.remove("hidden");
    }
  });
}

#### JavaScript Filter

```javascript
function setupFilters() {
  const filterToggle = document.getElementById("filter-toggle");
  const filterDropdown = document.getElementById("filter-dropdown");
  const starSlider = document.getElementById("star-slider");
  const starInput = document.getElementById("star-input");

  //Toggle dropdown
  filterToggle.addEventListener("click", function () {
    filterDropdown.classList.toggle("show");
    filterToggle.classList.toggle("active");
  });

  //Star rating yang tersinkronisasi
  starSlider.addEventListener("input", function () {
    starInput.value = this.value;
    updateStarDisplay(this.value);
    applyFilters();
  });

  //Pengaplikasian filter berdasarkan rating bintang dan pencarian
  function applyFilters() {
    const slides = document.querySelectorAll(".carousel-slide");
    const searchTerm = document
      .getElementById("food-search")
      .value.toLowerCase();
    const starFilter = parseInt(starSlider.value);

    slides.forEach((slide) => {
      const foodId = slide.dataset.food;
      const rating = getRating(foodId);
      const title = slide
        .querySelector(".carousel-title")
        .textContent.toLowerCase();

      const matchesSearch = title.includes(searchTerm);
      const matchesRating = starFilter === 0 || rating >= starFilter;
      const matchesFilter = shouldShowSlide(foodId);

      if (matchesSearch && matchesRating && matchesFilter) {
        slide.style.display = "block";
      } else {
        slide.style.display = "none";
      }
    });
  }
}
```

```javascript
function setupStarRating() {
  document.querySelectorAll(".star-rating").forEach((starRating) => {
    const foodId = starRating.getAttribute("data-food-id");
    if (foodId) {
      updateStarDisplay(foodId, getRating(foodId));
      setupStarClickHandler(starRating, foodId);
    }
  });
}

function setupStarClickHandler(starRating, foodId) {
  const stars = starRating.querySelectorAll(".star");

  stars.forEach((star) => {
    const rating = parseInt(star.getAttribute("data-rating"));

    star.addEventListener("click", (e) => {
      e.stopPropagation();
      setRating(foodId, rating);
    });

    star.addEventListener("mouseenter", () => {
      highlightStars(starRating, rating);
    });
  });

  starRating.addEventListener("mouseleave", () => {
    updateStarDisplay(foodId, getRating(foodId));
  });
}

//Integrasi hasil rating dengan local storage
function setRating(foodId, rating) {
  userRatings[foodId] = rating;
  localStorage.setItem("kulinerManadoRatings", JSON.stringify(userRatings));
  updateStarDisplay(foodId, rating);
  applyFilters();
}
```

function setupCarousel() {

function setupSearchSuggestions() {
  const searchInput = document.getElementById("food-search");
  const suggestionsContainer = document.getElementById("search-suggestions");

  let searchTimeout;

  searchInput.addEventListener("input", function () {
    clearTimeout(searchTimeout);
    const query = this.value.trim().toLowerCase();

    if (query.length === 0) {
      hideSuggestions();
      filterCarousel();
      return;
    }

    searchTimeout = setTimeout(() => {
      showSuggestions(query);
      filterCarousel();
    }, 150);
  });

  function showSuggestions(query) {
    const suggestions = Object.keys(foodData)
      .filter((foodId) => {
        const title = foodData[foodId].title.toLowerCase();
        return title.includes(query);
      })
      .slice(0, 5);

    if (suggestions.length === 0) {
      hideSuggestions();
      return;
    }

    const html = suggestions
      .map((foodId) => {
        const title = foodData[foodId].title;
        const highlightedTitle = highlightText(title, query);
        return `
                <div class="suggestion-item" data-food-id="${foodId}">
                    ${highlightedTitle}
                </div>
            `;
      })
      .join("");

    suggestionsContainer.innerHTML = html;
    suggestionsContainer.classList.add("show");

    //Bisa memilih saran pencarian
    suggestionsContainer
      .querySelectorAll(".suggestion-item")
      .forEach((item) => {
        item.addEventListener("click", function () {
          const foodId = this.dataset.foodId;
          searchInput.value = foodData[foodId].title;
          hideSuggestions();
          filterCarousel();
        });
      });
  }

  function highlightText(text, query) {
    const regex = new RegExp(`(${query})`, "gi");
    return text.replace(regex, "<mark>$1</mark>");
  }

  function hideSuggestions() {
    suggestionsContainer.classList.remove("show");
  }
}

//Inisialisasi burger menu hanya pada mode mobile
function setupBurgerMenu() {
  const burgerMenu = document.getElementById("burgerMenu");
  const navMenu = document.getElementById("navMenu");
  const burgerIcon = document.getElementById("burgerIcon");
  const closeIcon = document.getElementById("closeIcon");

  if (!burgerMenu || !navMenu) return;

  burgerMenu.addEventListener("click", function () {
    burgerMenu.classList.toggle("active");
    navMenu.classList.toggle("active");

    if (navMenu.classList.contains("active")) {
      burgerIcon.style.display = "none";
      closeIcon.style.display = "block";
      document.body.style.overflow = "hidden";
    } else {
      burgerIcon.style.display = "block";
      closeIcon.style.display = "none";
      document.body.style.overflow = "";
    }
  });

  //Tutup menu ketika salah satu link navigasi diklik
  document.querySelectorAll(".top-nav-item").forEach((item) => {
    item.addEventListener("click", function () {
      closeMenu();
    });
  });

  // Tutup menu ketika ukuran layar berubah (Desktop)
  window.addEventListener("resize", function () {
    if (window.innerWidth > 768) {
      closeMenu();
    }
  });

  function closeMenu() {
    burgerMenu.classList.remove("active");
    navMenu.classList.remove("active");
    burgerIcon.style.display = "block";
    closeIcon.style.display = "none";
    document.body.style.overflow = "";
  }
}
